SE A ROTA NÃO EXISTIR:
Adicionar ANTES de qualquer outra rota DELETE:
```typescript
// Exclusão em massa
router.delete('/bulk-delete', authenticateToken, async (req, res) => {
  console.log('🔴 BULK DELETE - Iniciando');
  console.log('Body:', req.body);
  
  const client = await pool.connect();
  
  try {
    const { ids } = req.body;
    const tenantId = req.user.tenantId;
    
    // Validações
    if (!ids || !Array.isArray(ids) || ids.length === 0) {
      return res.status(400).json({ error: 'IDs inválidos' });
    }
    
    console.log(`Deletando ${ids.length} registros do tenant ${tenantId}`);
    
    await client.query('BEGIN');
    
    // Deletar registros relacionados primeiro
    await client.query(
      `DELETE FROM import_records WHERE import_run_id = ANY($1::int[])`,
      [ids]
    );
    
    // Deletar import_runs
    const result = await client.query(
      `DELETE FROM import_runs WHERE id = ANY($1::int[]) AND tenant_id = $2 RETURNING id`,
      [ids, tenantId]
    );
    
    await client.query('COMMIT');
    
    console.log(`✅ ${result.rowCount} registros deletados`);
    
    res.json({
      success: true,
      deleted: result.rowCount,
      message: `${result.rowCount} base(s) excluída(s)`
    });
    
  } catch (error) {
    await client.query('ROLLBACK');
    console.error('❌ Erro bulk delete:', error);
    res.status(500).json({ 
      error: 'Erro ao excluir imports',
      message: error.message 
    });
  } finally {
    client.release();
  }
});
```
