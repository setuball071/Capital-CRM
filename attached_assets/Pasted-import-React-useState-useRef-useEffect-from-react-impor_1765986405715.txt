const AIChatPanel = ({ meeting, onClose }) => {
  const [messages, setMessages] = useState([{ role: 'assistant', text: `Olá! Analisei a ata "${meeting.title}". O que gostaria de saber?` }]);
  const [input, setInput] = useState('');
  const [loading, setLoading] = useState(false);
  const scrollRef = useRef(null);

  useEffect(() => { if (scrollRef.current) scrollRef.current.scrollTop = scrollRef.current.scrollHeight; }, [messages]);

  const handleSend = async () => {
    if (!input.trim()) return;
    const userMsg = { role: 'user', text: input };
    setMessages(prev => [...prev, userMsg]);
    setInput('');
    setLoading(true);

    try {
      const context = JSON.stringify({ title: meeting.title, summary: meeting.summary, decisions: meeting.decisions, risks: meeting.risks, participants: meeting.participants });
      const prompt = `Você é um assistente estratégico. Responda baseando-se EXCLUSIVAMENTE neste contexto: ${context}. Pergunta: "${userMsg.text}"`;
      const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/${MODEL_NAME}:generateContent?key=${apiKey}`, {
        method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] })
      });
      const data = await response.json();
      const aiText = data.candidates?.[0]?.content?.parts?.[0]?.text || "Desculpe, não consegui processar a resposta.";
      setMessages(prev => [...prev, { role: 'assistant', text: aiText }]);
    } catch (error) { setMessages(prev => [...prev, { role: 'assistant', text: "Erro de conexão com a IA." }]); } finally { setLoading(false); }
  };

  return (
    <div className="fixed right-0 top-0 h-full w-full md:w-96 bg-white shadow-2xl z-40 flex flex-col animate-slide-in-right border-l border-slate-200">
      <div className="p-4 border-b border-slate-100 bg-slate-900 text-white flex justify-between items-center"><div className="flex items-center gap-2"><Bot size={20} className="text-amber-400" /><h3 className="font-bold">Chat com Ata</h3></div><button onClick={onClose} className="p-1 hover:bg-slate-700 rounded transition-colors"><X size={20} /></button></div>

const NewMeetingAIModal = ({ onClose, onSave, templates }) => {
  const [notes, setNotes] = useState('');
  const [selectedTemplateId, setSelectedTemplateId] = useState(templates[0].id);
  const [loading, setLoading] = useState(false);
  const [step, setStep] = useState('input'); 

  const generateReport = async () => {
    if (!notes.trim()) return;
    setLoading(true);
    setStep('loading');

    const selectedTemplate = templates.find(t => t.id === selectedTemplateId);
    const fullPrompt = `${selectedTemplate.prompt} NOTAS BRUTAS DA REUNIÃO: "${notes}"`;

    try {
      const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/${MODEL_NAME}:generateContent?key=${apiKey}`, {
        method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: fullPrompt }] }] })
      });
      const data = await response.json();
      let generatedText = data.candidates?.[0]?.content?.parts?.[0]?.text;
      generatedText = generatedText.replace(/```json/g, '').replace(/```/g, '').trim();
      const parsedData = JSON.parse(generatedText);
      if (!parsedData.tags) parsedData.tags = [];
      parsedData.tags.push(selectedTemplate.label);
      onSave(parsedData);
    } catch (error) { console.error("Erro Gemini:", error); alert("Erro ao gerar relatório."); setStep('input'); } finally { setLoading(false); }
  };

  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center bg-slate-900/60 backdrop-blur-sm p-4 animate-fade-in">
      <div className="bg-white rounded-2xl shadow-2xl w-full max-w-2xl overflow-hidden flex flex-col max-h-[90vh]">

          <div className="p-6 grid grid-cols-1 md:grid-cols-2 gap-6">{analysis.map((person, index) => (<AIParticipantCard key={index} data={person} />))}</div>
        </div>
      ) : (<div className="bg-slate-50 p-6 rounded-xl border border-dashed border-slate-300 text-center text-slate-500"><Brain className="mx-auto mb-2 opacity-50" /><p>Análise de IA não disponível.</p></div>)}
      <div className="bg-white rounded-xl shadow-sm border border-slate-200 p-6">
        <h3 className="text-lg font-bold mb-4 text-slate-800">Overview</h3>
        <p className="text-slate-600 mb-4 leading-relaxed whitespace-pre-line">{meeting.summary}</p>
        <div className="flex flex-wrap gap-2 mt-4">{meeting.tags?.map((tag, i) => (<Tag key={i} label={tag} color={i % 2 === 0 ? "blue" : "purple"} />))}</div>
      </div>
    </div>
  );

  const renderTabContent = (activeTab, meetingData) => {
    const isDefaultMeeting = meetingData.id === 1;
    const analysisData = isDefaultMeeting ? defaultAiAnalysisData : (meetingData.aiAnalysis || []);
    switch (activeTab) {
      case 'resumo': return <ResumoTab meeting={meetingData} analysis={analysisData} />;
      case 'decisoes': return <DecisoesTab meeting={meetingData} />;
      case 'financeiro': return <FinanceiroTab />;
      case 'pessoas': return <PessoasTab />;
      case 'riscos': return <RiscosTab meeting={meetingData} />;
      case 'pendencias': return <PendenciasTab />;
      default: return <ResumoTab meeting={meetingData} analysis={analysisData} />;
    }
  };

  return (
    <div className="min-h-screen bg-slate-50 text-slate-800 font-sans flex flex-col md:flex-row overflow-hidden relative">
      <aside className="w-full md:w-72 bg-slate-900 text-white flex-shrink-0 flex flex-col shadow-xl z-20 transition-all duration-300">
        <div className="p-6 border-b border-slate-700 bg-slate-900">
          <div className="flex items-center gap-3">
            <div className={`w-10 h-10 rounded-xl flex items-center justify-center shadow-lg shadow-amber-500/20 ${activeMeeting ? 'bg-slate-800 text-slate-400' : 'bg-amber-500 text-slate-900'}`}><span className="font-bold text-xl">G</span></div>
            <div><h1 className="font-bold text-lg tracking-tight leading-none">GOLD GROUP</h1><p className="text-xs text-slate-400 mt-1">Strategic Hub</p></div>
          </div>
        </div>
        <div className="px-6 py-4 bg-slate-800/30 border-b border-slate-800 flex items-center gap-3">
           <div className="w-8 h-8 rounded-full bg-gradient-to-tr from-amber-400 to-amber-600 flex items-center justify-center font-bold text-slate-900 shadow-md">{currentUser.avatar}</div>
           <div className="flex-1 overflow-hidden"><p className="text-sm font-bold text-white truncate">{currentUser.name}</p><p className="text-xs text-slate-400 truncate flex items-center gap-1"><Shield size={10} className={currentUser.role === 'manager' ? 'text-purple-400' : 'text-blue-400'} />{currentUser.label}</p></div>
        </div>
        <nav className="flex-1 overflow-y-auto py-6 px-3 space-y-2">
          {!activeMeeting ? (
            <>
              <div className="px-4 pb-2 text-xs font-bold text-slate-500 uppercase tracking-wider">Menu Principal</div>
              <NavButton label="Todas as Reuniões" icon={<LayoutDashboard size={18} />} active={activeSection === 'meetings'} onClick={() => handleNavigate('meetings')} />
              <NavButton label="Equipe & Acessos" icon={<Users size={18} />} active={activeSection === 'team'} onClick={() => handleNavigate('team')} />
               {currentUser.role === 'manager' && (<div className="pt-4 mt-4 border-t border-slate-800"><div className="px-4 pb-2 text-xs font-bold text-slate-500 uppercase tracking-wider">Configuração</div><NavButton label="Prompts & IA" icon={<Settings size={18} />} active={false} onClick={() => setShowSettingsModal(true)} /></div>)}
            </>
          ) : (
            <>
              <button onClick={() => handleNavigate('meetings')} className="w-full flex items-center gap-2 px-4 py-3 mb-4 text-sm font-medium text-slate-400 hover:text-white hover:bg-slate-800 rounded-lg transition-colors border border-dashed border-slate-700"><ArrowLeft size={16} /> Voltar para Lista</button>
              <div className="px-4 pb-2 text-xs font-bold text-slate-500 uppercase tracking-wider">Navegação do Relatório</div>
              <NavButton id="resumo" label="Resumo & IA Insights" icon={<Brain size={18} />} active={activeTab === 'resumo'} onClick={setActiveTab} />
              <NavButton id="decisoes" label="Decisões Chave" icon={<Gavel size={18} />} active={activeTab === 'decisoes'} onClick={setActiveTab} />
              <NavButton id="financeiro" label="Financeiro & Projeções" icon={<TrendingUp size={18} />} active={activeTab === 'financeiro'} onClick={setActiveTab} />
              <NavButton id="pessoas" label="Pessoas & Estrutura" icon={<Users size={18} />} active={activeTab === 'pessoas'} onClick={setActiveTab} />
              <NavButton id="riscos" label="Riscos & Jurídico" icon={<ShieldAlert size={18} />} active={activeTab === 'riscos'} onClick={setActiveTab} />
              <NavButton id="pendencias" label="Pendências" icon={<CheckSquare size={18} />} active={activeTab === 'pendencias'} onClick={setActiveTab} />
              <div className="mt-6 px-3"><button onClick={() => setShowChat(!showChat)} className={`w-full flex items-center justify-center gap-2 py-3 rounded-xl text-sm font-bold transition-all shadow-lg ${showChat ? 'bg-amber-500 text-slate-900' : 'bg-slate-800 text-amber-500 border border-amber-500/30'}`}><Sparkles size={16} />{showChat ? 'Fechar Chat IA' : 'Chat com Ata'}</button></div>
            </>
          )}
        </nav>
        <UserSwitcher currentUser={currentUser} onSwitch={setCurrentUser} users={team} />
        <div className="p-4 bg-slate-800/50 border-t border-slate-800 backdrop-blur-sm space-y-2">
          {!activeMeeting ? currentUser.role === 'manager' && (<button onClick={() => setShowNewMeetingModal(true)} className="w-full flex items-center justify-center gap-2 bg-amber-500 hover:bg-amber-600 text-slate-900 py-2.5 px-4 rounded-lg text-sm font-bold transition-colors shadow-lg shadow-amber-500/20"><PlusCircle size={16} /> Nova Ata (IA)</button>) : (<div className="text-xs text-slate-400 space-y-1"><p className="font-medium text-white line-clamp-1">{activeMeeting.title}</p><div className="flex items-center gap-2"><Clock size={12} /> {activeMeeting.duration}</div></div>)}
          <button onClick={handleLogout} className="w-full flex items-center justify-center gap-2 bg-slate-800 hover:bg-red-900/30 text-slate-400 hover:text-red-400 py-2 px-4 rounded-lg text-xs font-bold transition-all border border-transparent hover:border-red-900/50"><LogOut size={14} /> Sair</button>
        </div>
      </aside>

      <main className="flex-1 overflow-y-auto bg-slate-50 relative">
        {selectedMeetingId ? (
          <div className="p-4 md:p-8 animate-fade-in pb-24 md:pb-8">
            <header className="mb-8 flex flex-col md:flex-row md:items-center justify-between gap-4">
              <div><div className="flex items-center gap-2 text-sm text-slate-500 mb-2"><span className="hover:underline cursor-pointer" onClick={() => handleNavigate('meetings')}>Reuniões</span><ChevronRight size={14} /><span>{activeMeeting.date}</span></div><h2 className="text-3xl font-bold text-slate-800 tracking-tight flex items-center gap-3">{activeTab === 'resumo' && 'Resumo & Insights IA'}{activeTab === 'decisoes' && 'Decisões e Definições'}{activeTab === 'financeiro' && 'Planejamento Financeiro'}{activeTab === 'pessoas' && 'Estrutura & Pessoas'}{activeTab === 'riscos' && 'Matriz de Riscos'}{activeTab === 'pendencias' && 'Próximos Passos'}</h2><p className="text-slate-500 mt-1 flex items-center gap-2"><span className="w-2 h-2 rounded-full bg-amber-500"></span>Projeto: {activeMeeting.project}</p></div>
              <div className="flex items-center gap-3"><span className={`px-3 py-1 rounded-full text-xs font-bold border ${activeMeeting.status === 'Validado' ? 'bg-green-100 text-green-700 border-green-200' : 'bg-blue-100 text-blue-700 border-blue-200'}`}>{activeMeeting.status}</span>{(currentUser.role === 'manager' || currentUser.id === activeMeeting.ownerId) && (<button onClick={() => requestDeleteMeeting(activeMeeting.id)} className="p-2 text-slate-400 hover:text-red-600 hover:bg-red-50 rounded-full transition-colors" title="Excluir esta ata"><Trash2 size={20} /></button>)}</div>
            </header>
            <div className="bg-white/50 rounded-2xl p-1">{renderTabContent(activeTab, activeMeeting)}</div>
          </div>
        ) : (
          <>
            {activeSection === 'meetings' && (<MeetingListView onSelectMeeting={handleSelectMeeting} meetings={filteredMeetings} onNewMeeting={() => setShowNewMeetingModal(true)} onDeleteMeeting={requestDeleteMeeting} currentUser={currentUser} />)}
            {activeSection === 'team' && (<TeamView team={team} currentUser={currentUser} onAddUserClick={openCreateUser} onEditUser={openEditUser} onDeleteUser={requestDeleteUser} onMassDelete={requestMassDeleteUsers} />)}
          </>
        )}
        {showNewMeetingModal && (<NewMeetingAIModal onClose={() => setShowNewMeetingModal(false)} onSave={handleAddMeeting} templates={templates} />)}
        {showUserModal && (<UserModal onClose={() => setShowUserModal(false)} onSave={handleSaveUser} user={editingUser} />)}
        {showSettingsModal && (<SettingsModal onClose={() => setShowSettingsModal(false)} templates={templates} onSave={handleSaveTemplates} />)}
        {deleteConfirmation.isOpen && (<DeleteConfirmationModal title={getDeleteMessage().title} text={getDeleteMessage().text} onClose={() => setDeleteConfirmation({ isOpen: false, type: null, id: null, ids: [] })} onConfirm={confirmDelete} />)}
        {activeMeeting && showChat && (<AIChatPanel meeting={activeMeeting} onClose={() => setShowChat(false)} />)}
      </main>
    </div>
  );
}
