const importarJSON = async () => {
  try {
    // Validar JSON
    const dados = JSON.parse(jsonInput);
    
    if (!dados.colaborador || !dados.cpf) {
      setErroJSON('JSON deve ter campos "colaborador" e "cpf"');
      return;
    }
    
    setJsonValido(true);
    setErroJSON(null);
    
    // Enviar para backend
    const response = await fetch('/api/employees/import-json', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: jsonInput
    });
    
    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.error || 'Erro ao importar');
    }
    
    // Sucesso
    toast.success('Funcionário importado com sucesso!');
    setShowImportModal(false);
    setJsonInput('');
    setJsonValido(false);
    
    // Recarregar lista
    refetch(); // ou window.location.reload();
    
  } catch (error) {
    if (error instanceof SyntaxError) {
      setErroJSON('JSON inválido. Verifique a formatação.');
    } else {
      setErroJSON(error.message);
    }
    setJsonValido(false);
  }
};
