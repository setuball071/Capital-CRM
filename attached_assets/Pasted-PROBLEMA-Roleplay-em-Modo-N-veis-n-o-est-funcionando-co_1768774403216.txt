```typescript
   async function avaliarSessaoRoleplay(sessaoId: number) {
     const sessao = await db.select()
       .from(roleplaySessoes)

     if (!sessao) throw new Error('Sessão não encontrada');

   async function avaliarComCriteriosDoNivel(sessao) {
     const nivel = sessao.nivelTreinado;
     
     // Buscar configuração do nível
     const nivelConfig = await db.select()
       .from(roleplayNivelPrompts)
       .where(eq(roleplayNivelPrompts.nivel, nivel))
       .where(eq(roleplayNivelPrompts.tenantId, sessao.tenantId))
       .limit(1);
     
     if (!nivelConfig) throw new Error(`Config do nível ${nivel} não encontrada`);
     
     // Montar prompt de avaliação ESPECÍFICO DO NÍVEL
     const promptAvaliacao = `
Você é um avaliador de treinamento de vendas de crédito consignado.

Analise esta sessão de roleplay do NÍVEL ${nivel}: ${nivelConfig.nome}.

CRITÉRIOS ESPECÍFICOS DESTE NÍVEL:
${nivelConfig.criteriosAprovacao.map((c, i) => `${i + 1}. ${c}`).join('\n')}

NOTA MÍNIMA PARA APROVAÇÃO: ${nivelConfig.notaMinima}/10

HISTÓRICO DA CONVERSA:
${JSON.stringify(sessao.historicoConversa)}

AVALIE e retorne APENAS um JSON válido (sem markdown, sem explicações):
{
  "notaGeral": 7.5,
  "aprovado": true,
  "pontosFortes": ["Ponto 1", "Ponto 2"],
  "pontosMelhorar": ["Melhoria 1", "Melhoria 2"],
  "criteriosAtendidos": {
    "criterio1": true,
    "criterio2": false,
    ...
  },
  "feedback": "Texto detalhado do feedback"
}
`;
     
     // Chamar OpenAI para avaliar
     const response = await openai.chat.completions.create({
       model: "gpt-4",
       messages: [
         { role: "user", content: promptAvaliacao }
       ],
       temperature: 0.3, // Mais determinístico para avaliação
     });
     
     // Parsear resposta
     const avaliacaoText = response.choices[0].message.content;
     const avaliacao = JSON.parse(avaliacaoText);
     
     // Verificar se passou baseado na nota mínima do nível
     const passou = avaliacao.notaGeral >= nivelConfig.notaMinima;
     
     // Salvar avaliação no banco
     await db.insert(roleplayAvaliacoes).values({
       sessaoId: sessao.id,
       userId: sessao.userId,
       nivelAvaliado: nivel,  // ✅ SALVAR qual nível foi avaliado
       notaGlobal: avaliacao.notaGeral,
       aprovadoProximoNivel: passou,  // ✅ CRÍTICO: define se desbloqueou
       pontosFortes: avaliacao.pontosFortes,
       pontosMelhorar: avaliacao.pontosMelhorar,
       comentarioGeral: avaliacao.feedback,
       criteriosAtendidos: avaliacao.criteriosAtendidos,
       createdAt: new Date(),
     });
     
     // ✅ SE PASSOU, ATUALIZAR NÍVEL DO VENDEDOR
     if (passou) {
       const vendedorAcademia = await db.select()
         .from(vendedoresAcademia)
         .where(eq(vendedoresAcademia.userId, sessao.userId))
         .limit(1);
       
       if (vendedorAcademia && vendedorAcademia.nivelAtual === nivel) {
         // Só avança se ainda estiver nesse nível (evita pular)
         await db.update(vendedoresAcademia)
           .set({ 
             nivelAtual: nivel + 1,
             atualizadoEm: new Date()
           })
           .where(eq(vendedoresAcademia.userId, sessao.userId));
       } else if (!vendedorAcademia) {
         // Se não existe registro, criar
         await db.insert(vendedoresAcademia).values({
           userId: sessao.userId,
           nivelAtual: nivel + 1,
           quizAprovado: false,
           criadoEm: new Date(),
           atualizadoEm: new Date(),
         });
       }
     }
     
     // Atualizar sessão como finalizada
     await db.update(roleplaySessoes)
       .set({ 
         status: 'finalizada',
         aprovado: passou,
         notaFinal: avaliacao.notaGeral,
         finalizadoEm: new Date()
       })
       .where(eq(roleplaySessoes.id, sessao.id));
     
     return {
       ...avaliacao,
       passou,
       nivelAtual: nivel,
       proximoNivel: passou ? nivel + 1 : nivel,
     };
   }
```

4. INTERFACE DE SELEÇÃO DEVE VERIFICAR PROGRESSO:

   Ao carregar tela de níveis:
```typescript
   async function carregarProgressoNiveis(userId: number) {
     // Buscar nível atual do vendedor
     const vendedor = await db.select()
       .from(vendedoresAcademia)
       .where(eq(vendedoresAcademia.userId, userId))
       .limit(1);
     
     const nivelAtual = vendedor?.nivelAtual || 1;
     
     // Buscar histórico de avaliações
     const avaliacoes = await db.select()
       .from(roleplayAvaliacoes)
       .where(eq(roleplayAvaliacoes.userId, userId))
       .orderBy(desc(roleplayAvaliacoes.createdAt));
     
     // Montar status de cada nível
     const niveis = [1, 2, 3, 4, 5].map(n => {
       const melhorAvaliacao = avaliacoes
         .filter(a => a.nivelAvaliado === n && a.aprovadoProximoNivel)
         .sort((a, b) => b.notaGlobal - a.notaGlobal)[0];
       
       return {
         nivel: n,
         status: n < nivelAtual ? 'concluido' : 
                 n === nivelAtual ? 'disponivel' : 'bloqueado',
         melhorNota: melhorAvaliacao?.notaGlobal,
         tentativas: avaliacoes.filter(a => a.nivelAvaliado === n).length,
       };
     });
     
     return niveis;
   }
```
