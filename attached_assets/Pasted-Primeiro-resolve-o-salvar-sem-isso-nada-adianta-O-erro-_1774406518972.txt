jsasync function savePDF() {
  if (!pages.length) { toast('Nenhum PDF aberto'); return; }
  load(true, 'Gerando PDF...');
  try {
    const { jsPDF } = window.jspdf;
    const first = pages[0];
    const orient = first.w_pt > first.h_pt ? 'l' : 'p';
    const doc = new jsPDF({ orientation: orient, unit: 'pt', format: [first.w_pt, first.h_pt] });

    for (let i = 0; i < pages.length; i++) {
      const p = pages[i];
      if (i > 0) doc.addPage([p.w_pt, p.h_pt], p.w_pt > p.h_pt ? 'l' : 'p');

      const pdfJsPage = await pdfJsDoc.getPage(p.num);
      const scale = 3;
      const vp = pdfJsPage.getViewport({ scale });
      const cv = document.createElement('canvas');
      cv.width = vp.width; cv.height = vp.height;
      const octx = cv.getContext('2d');
      await pdfJsPage.render({ canvasContext: octx, viewport: vp }).promise;

      const sx = vp.width / p.w_pt;
      const sy = vp.height / p.h_pt;

      const corrs = corrections.filter(c => c.page === p.num);
      for (const c of corrs) {
        const rx0 = c.bbox[0] * sx;
        const ry0 = c.bbox[1] * sy;
        const rw  = (c.bbox[2] - c.bbox[0]) * sx;
        const rh  = (c.bbox[3] - c.bbox[1]) * sy;

        let bgR = 255, bgG = 255, bgB = 255;
        try {
          const px = octx.getImageData(Math.max(0,Math.floor(rx0)), Math.max(0,Math.floor(ry0)), 1, 1).data;
          bgR = px[0]; bgG = px[1]; bgB = px[2];
        } catch(e) {}

        octx.fillStyle = `rgb(${bgR},${bgG},${bgB})`;
        octx.fillRect(rx0 - 2, ry0 - 2, rw + 4, rh + 4);

        if (!c.delete && c.new) {
          const hexC = c.color || '#000000';
          const cr = parseInt(hexC.slice(1,3),16);
          const cg = parseInt(hexC.slice(3,5),16);
          const cb = parseInt(hexC.slice(5,7),16);
          const fsize = c.size * sx;
          const fw = c.bold ? '700' : '400';
          const ff = c.font.includes('Times') ? 'Times New Roman' : c.font.includes('Courier') ? 'Courier New' : 'Arial';
          octx.font = `${fw} ${fsize}px ${ff}`;
          octx.fillStyle = `rgb(${cr},${cg},${cb})`;
          octx.fillText(c.new, rx0, ry0 + fsize * 0.85);
        }
      }

      const jpeg = cv.toDataURL('image/jpeg', 0.95);
      doc.addImage(jpeg, 'JPEG', 0, 0, p.w_pt, p.h_pt);
    }

    doc.save('corrigido.pdf');
    toast(`✓ PDF salvo com ${corrections.length} correção(ões)!`);
  } catch(e) {
    console.error(e);
    toast('Erro: ' + e.message);
  } finally {
    load(false);
  }
}
