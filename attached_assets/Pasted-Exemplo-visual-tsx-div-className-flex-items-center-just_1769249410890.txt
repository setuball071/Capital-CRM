7. PERSISTIR PREFERÊNCIA (OPCIONAL)
   Salvar escolha no localStorage:
```typescript
   const [itemsPerPage, setItemsPerPage] = useState(() => {
     const saved = localStorage.getItem('importHistoryItemsPerPage');
     return saved ? Number(saved) : 25;
   });
   
   useEffect(() => {
     localStorage.setItem('importHistoryItemsPerPage', String(itemsPerPage));
   }, [itemsPerPage]);
```

8. AJUSTE NO BACKEND (SE NECESSÁRIO)
   Se a paginação for server-side, atualizar endpoint:
```typescript
   // GET /api/import-runs?page=1&limit=25
   
   const page = parseInt(req.query.page as string) || 1;
   const limit = parseInt(req.query.limit as string) || 25;
   const offset = (page - 1) * limit;
   
   const results = await db
     .select()
     .from(importRuns)
     .where(eq(importRuns.tenantId, tenantId))
     .limit(limit)
     .offset(offset)
     .orderBy(desc(importRuns.createdAt));
   
   const totalCount = await db
     .select({ count: sql<number>`count(*)` })
     .from(importRuns)
     .where(eq(importRuns.tenantId, tenantId));
   
   res.json({
     data: results,
     pagination: {
       page,
       limit,
       total: totalCount[0].count,
       totalPages: Math.ceil(totalCount[0].count / limit)
     }
   });
```
