   function PromptGlobalManager() {

4. LÓGICA DE SELEÇÃO ALEATÓRIA AO INICIAR MODO LIVRE:
```typescript
   async function iniciarModoLivre(userId: number, cenarioCustomizado?: string) {
     // Buscar todas as variantes ativas
     const variantes = await db.select()
       .from(aiPrompts)
       .where(eq(aiPrompts.type, 'roleplay'))
       .where(eq(aiPrompts.scope, 'global'))
       .where(eq(aiPrompts.isActive, true));
     
     if (variantes.length === 0) {
       throw new Error('Nenhuma variante de prompt ativa encontrada');
     }
     
     // ✅ ESCOLHER ALEATORIAMENTE
     const varianteSorteada = variantes[Math.floor(Math.random() * variantes.length)];
     
     let systemPrompt = varianteSorteada.promptText;
     
     // Se usuário definiu cenário customizado, adicionar
     if (cenarioCustomizado) {
       systemPrompt += `\n\nCENÁRIO CUSTOMIZADO PELO USUÁRIO:\n${cenarioCustomizado}`;
     }
     
     // Criar sessão
     const sessao = await db.insert(roleplaySessoes).values({
       userId,
       modo: 'livre',
       status: 'ativa',
       cenario: cenarioCustomizado || `Persona: ${varianteSorteada.descricaoVariante}`,
       historicoConversa: [
         { role: 'system', content: systemPrompt }
       ],
       createdAt: new Date(),
     }).returning();
     
     return {
       sessao,
       personaSorteada: varianteSorteada.descricaoVariante
     };
   }
```

   async function avaliarModoLivre(sessao) {
     // Sistema atual - avaliação genérica
     // NÃO usa critérios específicos
     // NÃO afeta progressão de níveis
     
     const promptAvaliacao = `
Avalie esta sessão de roleplay de crédito consignado.

Forneça:
- Nota geral (0-10)
- Pontos fortes (3-5 pontos)
- Pontos a melhorar (3-5 pontos)
- Feedback geral

Histórico: ${JSON.stringify(sessao.historicoConversa)}

Retorne JSON:
{
  "notaGeral": 8.5,
  "pontosFortes": ["...", "..."],
  "pontosMelhorar": ["...", "..."],
  "feedback": "..."
}
`;
     
     const response = await openai.chat.completions.create({
       model: "gpt-4",
       messages: [{ role: "user", content: promptAvaliacao }],
       temperature: 0.3,
     });
     
     const avaliacao = JSON.parse(response.choices[0].message.content);
     
     // Salvar avaliação (sem nivel_avaliado, sem aprovado_proximo_nivel)
     await db.insert(roleplayAvaliacoes).values({
       sessaoId: sessao.id,
       userId: sessao.userId,
       notaGlobal: avaliacao.notaGeral,
       pontosFortes: avaliacao.pontosFortes,
       pontosMelhorar: avaliacao.pontosMelhorar,
       comentarioGeral: avaliacao.feedback,
       createdAt: new Date(),
     });
     
     return avaliacao;
   }
