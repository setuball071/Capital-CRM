5. LÓGICA DE SELEÇÃO DE PROMPT NO BACKEND:
```typescript
   async function iniciarRoleplayComPromptCorreto(
     userId: number, 
     modo: 'livre' | 'niveis',
     nivel?: number,
     cenarioCustomizado?: string
   ) {
     let systemPrompt: string;
     
     if (modo === 'livre') {
       // USA SISTEMA ATUAL (ai_prompts - global ou equipe)
       const userTeam = await getUserTeam(userId);
       
       const prompt = await db.select()
         .from(aiPrompts)
         .where(eq(aiPrompts.type, 'roleplay'))
         .where(
           or(
             and(eq(aiPrompts.scope, 'team'), eq(aiPrompts.teamId, userTeam.id)),
             eq(aiPrompts.scope, 'global')
           )
         )
         .where(eq(aiPrompts.isActive, true))
         .orderBy(desc(aiPrompts.updatedAt))
         .limit(1);
       
       systemPrompt = prompt?.promptText || DEFAULT_ROLEPLAY_PROMPT;
       
       // Adiciona cenário customizado do usuário
       if (cenarioCustomizado) {
         systemPrompt += `\n\nCENÁRIO CUSTOMIZADO PELO USUÁRIO:\n${cenarioCustomizado}`;
       }
       
     } else if (modo === 'niveis') {
       // USA NOVO SISTEMA (roleplay_nivel_prompts)
       const nivelPrompt = await db.select()
         .from(roleplayNivelPrompts)
         .where(eq(roleplayNivelPrompts.nivel, nivel))
         .where(eq(roleplayNivelPrompts.tenantId, tenantId))
         .where(eq(roleplayNivelPrompts.isActive, true))
         .limit(1);
       
       if (!nivelPrompt) {
         throw new Error(`Prompt do nível ${nivel} não encontrado`);
       }
       
       systemPrompt = nivelPrompt.promptCompleto;
     }
     
     // Criar sessão
     const sessao = await db.insert(roleplaySessoes).values({
       userId,
       modo,
       nivelTreinado: nivel,
       status: 'ativa',
       historicoConversa: [
         { role: 'system', content: systemPrompt }
       ]
     }).returning();
     
     return sessao;
   }
```
