const RESPONDENTE_SYSTEM_PROMPT = `
Você é um especialista em crédito consignado que responde de forma CLARA, DIRETA e HUMANA
sobre roteiros bancários.

Você receberá:
1) A consulta original do usuário (texto livre).
2) Uma lista de roteiros bancários em JSON, com campos como banco, convenio, segmento,
   tipo_operacao e dados (faixas_idade, publico_alvo, etc.).

Sua missão:
- Explicar rapidamente o que É POSSÍVEL fazer para esse caso,
  usando SOMENTE as informações dos roteiros enviados.
- Nunca invente banco, convênio, prazo ou regra que não estejam nos dados.
- Se a informação não estiver clara nos roteiros, diga isso explicitamente.

Estilo da resposta:
- Frases curtas, objetivas.
- Tom de consultor humano.
- Máximo de 3 parágrafos curtos.
- Se faltar informação para cravar a resposta, peça uma especificação adicional
  (por exemplo: "me diga o convênio: GOV SP, SIAPE, INSS...").

`;

3. Rota juntando tudo (interpretador + busca + resposta humana)

Exemplo simplificado em server/routes.ts:

import express from "express";
import { interpretarConsulta } from "./interpretadorConsulta";
import { openai } from "./openaiClient";
// importa função que busca no banco
import { buscarRoteirosPorFiltros } from "./roteirosService";

const router = express.Router();

router.get("/api/roteiros/ia-search", async (req, res) => {
  try {
    const q = String(req.query.q || "").trim();
    if (!q) return res.status(400).json({ error: "Consulta vazia" });

    // 1) interpreta a consulta
    const filtros = await interpretarConsulta(q);

    // 2) busca roteiros no Postgres com base nos filtros
    const roteiros = await buscarRoteirosPorFiltros(filtros);
    // (aqui roteiros é um array de objetos do banco)

    // 3) chama IA para montar resposta em linguagem natural
    const completion = await openai.responses.create({
      model: "gpt-4.1-mini",
      input: [
        {
          role: "system",
          content: RESPONDENTE_SYSTEM_PROMPT
        },
        {
          role: "user",
          content: `
Consulta do usuário: "${q}"

Filtros interpretados: ${JSON.stringify(filtros, null, 2)}

Roteiros encontrados (JSON):
${JSON.stringify(roteiros, null, 2)}
          `
        }
      ]
    });

    const respostaTexto = completion.output[0].content[0].text;

    return res.json({
      filtros,
      roteiros,
      resposta: respostaTexto
    });
  } catch (err: any) {
    console.error(err);
    return res.status(500).json({ error: "Erro na pesquisa inteligente" });
  }
});

export default router;
